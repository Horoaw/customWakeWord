# Upstream microWakeWord — the canonical training framework.
# Pins TensorFlow 2.16+, requires Python 3.10.
#
# NOTE: installed via `pip install -e /opt/microwakeword` in the Dockerfile,
# not from PyPI. Upstream's setup.py uses find_packages() but the audio/
# and layers/ subdirs ship without __init__.py, so a normal install drops
# `microwakeword.audio.augmentation` (which our build_features.py needs).
# The Dockerfile creates those __init__.py files before the editable install.

# Pinned by microwakeword itself, but listed here so `pip install -r` on a
# bare environment doesn't try to resolve a wildcard:
tensorflow>=2.16
# CuDNN pin: TF 2.21 ships with `nvidia-cudnn-cu12==9.1.0.70` in its
# transitive constraints but was actually compiled against 9.3+, producing:
#   "Loaded runtime CuDNN library: 9.1.0 but source was compiled with: 9.3.0"
# Pin explicitly above the 9.3 floor so the constraint resolves correctly.
nvidia-cudnn-cu12>=9.3,<10
ai-edge-litert            # tflite runtime (for inference + eval)
pymicro-features          # C audio_microfrontend op (must match runtime)
mmap-ninja                # RaggedMmap feature store used by upstream
audiomentations>=0.36     # CPU augmentation (RIR, noise, EQ, codec)
webrtcvad-wheels          # VAD silence removal in piper-sample-generator path

# piper-sample-generator deps (cloned at runtime; these are its Python deps).
#
# CUDA pin: RunPod's A100 SXM hosts (and most server-class hosts as of
# 2026-05) ship CUDA driver 12.7. PyTorch's default PyPI wheel for torch>=2.7
# is built with CUDA 13 and refuses to use a driver older than 13.x — it
# silently falls back to CPU with a warning ("The NVIDIA driver on your
# system is too old (found version 12070)"). 10k TTS samples on CPU at
# A100 hourly rates is a catastrophic burn.
#
# Pinning to torch 2.5.1 + torchaudio 2.5.1 from the cu124 index keeps
# us on a wheel built for CUDA 12.4 (compatible with driver 12.7+).
--extra-index-url https://download.pytorch.org/whl/cu124
torch==2.5.1+cu124
torchaudio==2.5.1+cu124
piper-phonemize-cross==1.2.1
